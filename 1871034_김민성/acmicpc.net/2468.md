# [안전 영역](https://www.acmicpc.net/problem/2468)

깊이우선탐색을 이용하여 해결했습니다. `traverse` 함수에서 `DIR` 상수를 활용하여 재귀적으로 방문하는게 주요 포인트입니다.

```cpp
#include <iostream>
#include <map>
#include <cstring>
#include <vector>
#include <queue>
#include <cmath>

using namespace std;

const int INF = 987654321;
const int MAX_N = 101;

const int DIR[4][2] = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};

int N;
int board[MAX_N][MAX_N];
bool visited[MAX_N][MAX_N];

bool inRange(int y, int x) {
    return 0 <= y && y < N && 0 <= x && x < N;
}

bool isSafeArea(int y, int x, int waterHeight) {
    return board[y][x] > waterHeight;
}

// 주변에 인접한 잠기지 않은 도시를 방문 처리한다.
void traverse(int y, int x, int waterHeight) {
    visited[y][x] = true;
    for (int i = 0; i < 4; ++i) {
        int ny = y + DIR[i][0];
        int nx = x + DIR[i][1];

        if (!inRange(ny, nx) || visited[ny][nx] || !isSafeArea(ny, nx, waterHeight)) {
            continue;
        }

        traverse(ny, nx, waterHeight);
    }
}

int solve(int maxHeight) {
    int ret = 1;

    for (int h = 1; h <= maxHeight; ++h) {
        int areaCount = 0;
        memset(visited, false, sizeof(visited));

        for (int y = 0; y < N; ++y) {
            for (int x = 0; x < N; ++x) {
                if (visited[y][x] || !isSafeArea(y, x, h)) {
                    continue;
                }
                ++areaCount;
                traverse(y, x, h);
            }
        }
        ret = max(areaCount, ret);
    }
    return ret;
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cout.tie(NULL);

    int maxHeight = 0;

    cin >> N;
    for (int i = 0; i < N; ++i) {
        for (int j = 0; j < N; ++j) {
            cin >> board[i][j];
            maxHeight = max(board[i][j], maxHeight);
        }
    }

    cout << solve(maxHeight) << endl;

    return 0;
}
```
